import java.awt.*;

class awt12 extends Frame {
    public static void main(String[] args) {
        Frame f = new Frame();
        Checkbox ctrl2 = new Checkbox("Male");
        Checkbox ctrl3 = new Checkbox("Female");
        CheckboxGroup chg = new CheckboxGroup();
        ctrl2.setCheckboxGroup(chg);
        ctrl3.setCheckboxGroup(chg);
        f.setLayout(new FlowLayout(FlowLayout.LEFT));
        f.setBounds(50, 50, 300, 300);
        f.add(ctrl2);
        f.add(ctrl3);
        f.setTitle("MyNotepad");
        f.setVisible(true);
    }
}
